import Title from "../../components/Title"

### 大小

使用 `size` 调整大小

<br/>

<Title size="ty">小小</Title>
<Title size="sm">小</Title>
<Title>不小</Title>
<Title size="lg">大</Title>
<Title size="xl">大大</Title>

```tsx copy
import { Title } from "mmixel-ui"

export default function App() {
  return (<>
    <Title size="ty">小小</Title>
    <Title size="sm">小</Title>
    <Title>不小</Title>
    <Title size="lg">大</Title>
    <Title size="xl">大大</Title>
  </>)
}
```

### Title Props

| Prop | Type                                   | Default | Desc       |
| :--- | :------------------------------------- | :------ | :--------- |
| size | `"ty" \| "sm" \| "md" \| "lg" \| "xl"` | `"md"`  | 标题的大小 |
