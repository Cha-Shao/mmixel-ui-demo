}

export const CarouselExample = async () => {
  const images = await getImages()

  return (
    <div className="relative">
      <Carousel>

  )

### Carousel Props

| Prop     | Type     | Default     | Desc                       |
| :------- | :------- | :---------- | :------------------------- |
| page     | `number` | `undefined` | 当前轮播页(index, 从0开始) |
| delay    | `number` | `0`         | 轮播播放前摇(ms)           |
| duration | `number` | `300`       | 轮播动画时长(ms)           |
